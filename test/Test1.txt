import React, { useState } from "react";
import Chatbot from "react-chatbot-kit";
import "react-chatbot-kit/build/main.css";

import config from "./chatbot/config";
import MessageParser from "./chatbot/MessageParser";
import ActionProvider from "./chatbot/ActionProvider";

function App() {
  const [showBot, setShowBot] = useState(false);

  const chatbotStyles = {
    width: "500px",
    height: "600px",
    border: "2px solid #007bff",
    borderRadius: "12px",
    backgroundColor: "#f9f9f9",
    padding: "10px",
    boxShadow: "0 4px 12px rgba(0,0,0,0.15)",
    position: "fixed",
    bottom: "80px",
    right: "20px",
    zIndex: 1000,
  };

  const buttonStyle = {
    position: "fixed",
    bottom: "20px",
    right: "20px",
    backgroundColor: "#007bff",
    color: "#fff",
    border: "none",
    borderRadius: "50%",
    width: "60px",
    height: "60px",
    fontSize: "24px",
    cursor: "pointer",
    boxShadow: "0 4px 8px rgba(0,0,0,0.2)",
  };

  const closeButtonStyle = {
    position: "absolute",
    top: "5px",
    right: "10px",
    background: "none",
    border: "none",
    fontSize: "20px",
    cursor: "pointer",
    color: "#333",
  };

  return (
    <>
      {/* Floating chat button (only shows if bot is hidden) */}
      {!showBot && (
        <button style={buttonStyle} onClick={() => setShowBot(true)}>
          üí¨
        </button>
      )}

      {/* Chatbot box */}
      {showBot && (
        <div style={chatbotStyles}>
          {/* Close button inside chatbot */}
          <button style={closeButtonStyle} onClick={() => setShowBot(false)}>
            ‚ùå
          </button>

          <Chatbot
            config={config}
            messageParser={MessageParser}
            actionProvider={ActionProvider}
          />
        </div>
      )}
    </>
  );
}

export default App;


import { createChatBotMessage } from "react-chatbot-kit";
import CustomMessage from "./CustomMessage";

const config = {
  botName: "MetsBot",
  customComponents: {
    botMessageBox: (props) => <CustomMessage {...props} isBot={true} />,
    userChatMessage: (props) => <CustomMessage {...props} isBot={false} />,
  },
  initialMessages: [
    createChatBotMessage("Hi! I can help you. üëã"),
  ],
};

export default config;

import React from "react";

const CustomMessage = ({ message, isBot }) => {
  const messageStyle = {
    maxWidth: "80%",
    padding: "10px 14px",
    margin: "8px 0",
    borderRadius: "12px",
    lineHeight: "1.4",
    fontSize: "15px",
    color: isBot ? "#fff" : "#333",
    backgroundColor: isBot ? "#007bff" : "#e6e6e6",
    alignSelf: isBot ? "flex-start" : "flex-end",
  };

  return <div style={messageStyle}>{message}</div>;
};

export default CustomMessage;
