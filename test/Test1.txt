Sub ExportRangeToPDF()
    Dim ws As Worksheet
    Dim rng As Range
    Dim pdfPath As String
    
    ' Set the worksheet and range to export
    Set ws = ThisWorkbook.Sheets("Sheet1") ' Change to your desired sheet name
    Set rng = ws.Range("A1:D20") ' Change to your desired range
    
    ' Define the output PDF file path
    pdfPath = ThisWorkbook.Path & "\ExportedRange.pdf" ' Saves in the same folder as the workbook
    
    ' Export the range as a PDF
    On Error Resume Next
    rng.ExportAsFixedFormat Type:=xlTypePDF, _
                            Filename:=pdfPath, _
                            Quality:=xlQualityStandard, _
                            IncludeDocProperties:=True, _
                            IgnorePrintAreas:=False, _
                            OpenAfterPublish:=True
                            
    ' Confirmation message
    MsgBox "PDF successfully created at: " & pdfPath, vbInformation, "Export Complete"
End Sub
