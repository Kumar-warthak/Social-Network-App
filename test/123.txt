<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8" />
  <title>SQL Display</title>
  <link
    href="https://cdn.jsdelivr.net/npm/bootstrap@5.3.2/dist/css/bootstrap.min.css"
    rel="stylesheet"
  />
  <script>
    async function fetchData() {
      const response = await fetch('http://localhost:5000/api/sqldata', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ key: 'sqldata' }),
      });
      const data = await response.json();
      renderJson(data, document.getElementById('output'));
      document.getElementById('sqlTextarea').value = ''; // Display only
    }

    function renderJson(obj, container, level = 0) {
      container.innerHTML = ''; // Clear previous
      const recurse = (data, parent) => {
        Object.entries(data).forEach(([key, value], i) => {
          const id = `collapse${level}_${key}_${i}`;
          const wrapper = document.createElement('div');
          wrapper.classList.add('mb-2');

          const header = document.createElement('h3');
          header.innerText = key;
          parent.appendChild(header);

          if (typeof value === 'object' && value !== null) {
            const btn = document.createElement('button');
            btn.className = 'btn btn-sm btn-outline-secondary mb-1';
            btn.setAttribute('data-bs-toggle', 'collapse');
            btn.setAttribute('data-bs-target', `#${id}`);
            btn.innerText = 'Toggle';
            parent.appendChild(btn);

            const nested = document.createElement('div');
            nested.className = 'collapse ms-3';
            nested.id = id;
            parent.appendChild(nested);

            recurse(value, nested);
          } else {
            const p = document.createElement('p');
            p.innerText = value;
            parent.appendChild(p);
          }
        });
      };

      recurse(obj, container);
    }
  </script>
</head>
<body class="container py-4">
  <h1 class="mb-4">SQL JSON Viewer</h1>

  <textarea id="sqlTextarea" class="form-control mb-3" rows="4" readonly></textarea>
  <button class="btn btn-primary mb-3" onclick="fetchData()">Fetch SQL Data</button>

  <div id="output"></div>

  <script src="https://cdn.jsdelivr.net/npm/bootstrap@5.3.2/dist/js/bootstrap.bundle.min.js"></script>
</body>
</html>
