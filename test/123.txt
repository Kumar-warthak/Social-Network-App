  <script>
    async function fetchData() {
      const response = await fetch('http://localhost:5000/sql_analysis', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json'
        },
        body: JSON.stringify({ sqldata: "your query or ID" })
      });

      const data = await response.json();

      const tbody = document.getElementById('sqlBody');
      tbody.innerHTML = ''; // Clear old data

      // Assume data is a list of records like mock_sql_analysis.json
      data.forEach(row => {
        const tr = document.createElement('tr');

        const tdLogic = document.createElement('td');
        tdLogic.textContent = row.query_logic;
        tr.appendChild(tdLogic);

        const tdSchema = document.createElement('td');
        tdSchema.textContent = row.schema;
        tr.appendChild(tdSchema);

        const tdTables = document.createElement('td');
        tdTables.innerHTML = row.table_with_columns.map(t => t.table).join('<br>');
        tr.appendChild(tdTables);

        const tdColumns = document.createElement('td');
        tdColumns.innerHTML = row.table_with_columns.map(t => t.columns.join(', ')).join('<br>');
        tr.appendChild(tdColumns);

        const tdJoins = document.createElement('td');
        tdJoins.textContent = row.joins;
        tr.appendChild(tdJoins);

        const tdWhere = document.createElement('td');
        tdWhere.textContent = row.where_clause;
        tr.appendChild(tdWhere);

        tbody.appendChild(tr);
      });
    }
  </script>
