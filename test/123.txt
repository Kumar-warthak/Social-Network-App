<!DOCTYPE html>
<html>
<head>
    <title>SQL Query Analysis</title>
    <style>
        table {
            border-collapse: collapse;
            width: 100%;
        }
        td, th {
            border: 1px solid black;
            padding: 8px;
            vertical-align: top;
        }
        th {
            background-color: #f2f2f2;
        }
    </style>
</head>
<body>
    <h2>SQL Query Analysis</h2>
    <table id="sql-table">
        <thead>
            <tr>
                <th>Query Logic</th>
                <th>Schema</th>
                <th colspan="2">Table with Columns</th>
                <th>Joins</th>
                <th>Where Statement</th>
            </tr>
            <tr>
                <th></th>
                <th></th>
                <th>Tables</th>
                <th>Columns</th>
                <th></th>
                <th></th>
            </tr>
        </thead>
        <tbody id="sql-body">
        </tbody>
    </table>

    <script>
        fetch('http://localhost:5000/api/sqldata', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ key: 'sqldata' })
        })
        .then(res => res.json())
        .then(data => {
            const body = document.getElementById('sql-body');
            const row = document.createElement('tr');

            const logic = document.createElement('td');
            logic.textContent = data.query_logic;
            row.appendChild(logic);

            const schema = document.createElement('td');
            schema.textContent = data.schema;
            row.appendChild(schema);

            // Tables
            const tables = document.createElement('td');
            tables.innerHTML = data.tables.map(t => `<div>${t.name}</div>`).join('');
            row.appendChild(tables);

            // Columns
            const columns = document.createElement('td');
            columns.innerHTML = data.tables.map(t => `<div>${t.columns.join(', ')}</div>`).join('');
            row.appendChild(columns);

            const joins = document.createElement('td');
            joins.textContent = data.joins;
            row.appendChild(joins);

            const where = document.createElement('td');
            where.textContent = data.where;
            row.appendChild(where);

            body.appendChild(row);
        })
        .catch(err => console.error('API error:', err));
    </script>
</body>
</html>
