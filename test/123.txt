<script>
  async function fetchScriptData() {
    const response = await fetch('https://your-api-url.com/analyze', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ sqldata: 'your-script-task-code-here' })
    });

    const json = await response.json();

    // Update Code Summary
    document.getElementById('code-summary').textContent = json["Code Summary"] || "No summary provided.";

    // Update Variables
    const varsBody = document.getElementById('variables-body');
    varsBody.innerHTML = "";
    (json.Variables || []).forEach(v => {
      const row = `<tr><td>${v["Variable Name"]}</td><td>${v["Usage"]}</td></tr>`;
      varsBody.innerHTML += row;
    });

    // Update Methods Accordion
    const methodsAccordion = document.getElementById('methodsAccordion');
    methodsAccordion.innerHTML = "";
    (json.Methods || []).forEach((method, index) => {
      const id = `method-${index}`;
      const item = `
        <div class="accordion-item">
          <h2 class="accordion-header" id="heading-${id}">
            <button class="accordion-button ${index !== 0 ? 'collapsed' : ''}" type="button" data-bs-toggle="collapse" data-bs-target="#collapse-${id}" aria-expanded="${index === 0}" aria-controls="collapse-${id}">
              ${method.Method || 'Unnamed Method'}
            </button>
          </h2>
          <div id="collapse-${id}" class="accordion-collapse collapse ${index === 0 ? 'show' : ''}" aria-labelledby="heading-${id}" data-bs-parent="#methodsAccordion">
            <div class="accordion-body">
              ${method.Explanation || 'No explanation provided.'}
            </div>
          </div>
        </div>
      `;
      methodsAccordion.innerHTML += item;
    });
  }

  fetchScriptData();
</script>
