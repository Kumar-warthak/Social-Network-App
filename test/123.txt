<body>
  <h2>SQL Analysis Result</h2>
  <div id="output"></div>

  <script>
    async function fetchAndDisplay() {
      const res = await fetch('http://127.0.0.1:5000/analyze-sql', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ query: 'your sql goes here' })
      });

      const data = await res.json();
      const outputDiv = document.getElementById('output');

      data.forEach((item, index) => {
        // Query Logic and Schema
        const infoSection = document.createElement('div');
        infoSection.className = 'section';
        infoSection.innerHTML = `
          <h3>Query #${index + 1}</h3>
          <p><strong>Query Logic:</strong> ${item.queryLogic}</p>
          <p><strong>Schema:</strong> ${item.schema}</p>
          <p><strong>Joins:</strong> ${item.joins}</p>
          <p><strong>Where:</strong> ${item.where}</p>
        `;
        outputDiv.appendChild(infoSection);

        // Tables and Columns
        const tableSection = document.createElement('div');
        tableSection.className = 'section';
        tableSection.innerHTML = `<h4>Tables and Columns</h4>`;

        const table = document.createElement('table');
        table.innerHTML = `
          <thead>
            <tr>
              <th>Table</th>
              <th>Columns</th>
            </tr>
          </thead>
          <tbody>
            ${Object.entries(item.tables).map(([tableName, columns]) => `
              <tr>
                <td>${tableName}</td>
                <td>${columns.join(', ')}</td>
              </tr>
            `).join('')}
          </tbody>
        `;
        tableSection.appendChild(table);
        outputDiv.appendChild(tableSection);
      });
    }

    fetchAndDisplay();
  </script>
</body>
