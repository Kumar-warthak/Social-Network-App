from decimal import Decimal

def is_valid_numeric(val, precision, scale):
    try:
        d = Decimal(str(val).strip())
        int_digits = len(d.as_tuple().digits) - d.as_tuple().exponent
        return int_digits <= (precision - scale)
    except:
        return False
