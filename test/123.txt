import re
import json

# Sample JSON-like string (from the webpage)
json_like = """{'name': 'John', 'age': 30, 'bio': 'Loves coding in \"Python\" & reading.'}"""

# Step 1: Replace single quotes around keys and values with double quotes
def fix_quotes(json_like):
    # Fix keys: 'key': → "key":
    json_like = re.sub(r"(?<=\{|\s)'(\w+)':", r'"\1":', json_like)

    # Fix string values: "key": 'value' → "key": "value"
    json_like = re.sub(r':\s*\'([^\']*)\'', r': "\1"', json_like)

    return json_like

fixed = fix_quotes(json_like)

# Step 2: Load as JSON
try:
    parsed = json.loads(fixed)
    print(parsed)
except json.JSONDecodeError as e:
    print("JSON decode error:", e)

# Step 3: Optional – write to file
with open("output.json", "w", encoding="utf-8") as f:
    json.dump(parsed, f, indent=2)
