    function populateValues() {
      const val1 = document.getElementById('input1').value;
      const val2 = document.getElementById('input2').value;

      document.getElementById('input3').value = `Computed: ${val1} + ${val2}`;
      document.getElementById('input4').value = val1.length + val2.length;
      document.getElementById('input5').value = val1.toUpperCase();
      document.getElementById('input6').value = val2.toLowerCase();
      document.getElementById('input7').value = new Date().toLocaleDateString();
      document.getElementById('input8').value = new Date().toLocaleTimeString();
    }
